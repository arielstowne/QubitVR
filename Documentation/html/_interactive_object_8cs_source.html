69  //-------------------------------------------------
70  // Called every Update() while a Hand is hovering over this object
71  //-------------------------------------------------
72  private void HandHoverUpdate(Hand hand)
73  {
74  GrabTypes startingGrabType = hand.GetGrabStarting();
75  bool isGrabEnding = hand.IsGrabEnding(this.gameObject);
76 
77  // User can only grab the object with the grip button.
78  // This was chosen because the trigger button triggers the laserpointer and causes unpleasant flashes when grabbing objects.
79  if (interactable.attachedToHand == null && startingGrabType == GrabTypes.Grip) // startingGrabType != GrabTypes.None
80  {
81  // Save our position/rotation so that we can restore it when we detach
82  oldPosition = transform.localPosition;
83  oldRotation = transform.localRotation;
84 
85  // Rotate the object so it aligns with the hand before it attaches.
86  transform.forward = hand.transform.forward;
87 
88  // Call this to continue receiving HandHoverUpdate messages,
89  // and prevent the hand from hovering over anything else
90  hand.HoverLock(interactable);
91 
92  // Attach this object to the hand
93  hand.AttachObject(gameObject, startingGrabType, attachmentFlags);
94  }
95  else if (isGrabEnding)
96  {
97  // If the gate gameObject is in contact with a qubit when the user lets go of it, apply the gate.
98  if (qubitTriggered != null)
99  {
100  // Applying gates still works the way it is currently, so this isn't a bug but more of a design choice:
101  // If you don't like that a gate can still be applied while the preview/animation for another gate 'plays',
102  // (applying the other gate just cuts the animation off) then use this in an if statement:
103  // GameObject.ReferenceEquals(gameObject, qubitTriggered.GetComponent<QubitTriggers>().TriggerObject)
104 
105  // Move the vector immediately and set the state.
106  qubitTriggered.GetComponent<ApplyGate>().ToolsInBounds(gameObject);
107  // We don't want the vector to return to the rotation it had before the gate entered the qubit.
108  // This would happen because QubitTriggers resets the rotation in onTriggerExit().
109  qubitTriggered.GetComponent<QubitTriggers>().ResetRotation = false;
110 
111  // If is a gate, play gate applied sound.
112  if (!checkGateType().Equals("flashlightStatus"))
113  {
114  audioSource.clip = gateApplied;
115  audioSource.Play();
116  }
117  }
118 
119  // Detach this object from the hand
120  hand.DetachObject(gameObject);
121  // Call this to undo HoverLock
122  hand.HoverUnlock(interactable);
123 
124  // Restore position/rotation
125  transform.localPosition = oldPosition;
126  transform.localRotation = oldRotation;
127  }
128  }

169  private string checkGateType()

172  return "flashlightStatus";
