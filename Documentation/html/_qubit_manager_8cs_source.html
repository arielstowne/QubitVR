24  public void SetRandomState(int n)
25  {
26  if (checkQubitExists(n))
27  {
28  // Here we specify the probabilities for each possible set of states, as stated above function.
29  // Lower hemisphere chance is 100 - (upChance + downChance + equatorChance + upperHemisphereChance).
30  int upChance = 10, downChance = 10, equatorChance = 20, upperHemisphereChance = 30;
31  float choice = Random.Range(0, 100);
32 
33  // If random state is up, set state to up and update assessment answer to "up".
34  if(choice < upChance)
35  {
36  qubitScript[n].setState(States.UP);
37  qubitScript[n].assessmentAnswer = "up";
38  }
39 
40  // If random state is down, set state to down and update assessment answer to "down".
41  else if (choice < upChance + downChance)
42  {
43  qubitScript[n].setState(States.DOWN);
44  qubitScript[n].assessmentAnswer = "down";
45  }
46 
47  // If random state is along the equator, set state to that and update update assessment answer to "equator".
48  else if (choice < upChance + downChance + equatorChance)
49  {
50  SetRandomEquatorQubit(qubits[n], qubitScript[n]);
51  qubitScript[n].assessmentAnswer = "equator";
52  }
53 
54  // If random state is likely up, set state to upper hemisphere and update update assessment answer to "likely_up".
55  else if (choice < upChance + downChance + equatorChance + upperHemisphereChance)
56  {
57  SetRandomHemisphereQubit(qubits[n], qubitScript[n], true);
58  qubitScript[n].assessmentAnswer = "likely_up";
59  }
60 
61  // If random state is likely down, set state to lower hemisphere and update update assessment answer to "likely_down".
62  else
63  {
64  SetRandomHemisphereQubit(qubits[n], qubitScript[n], false);
65  qubitScript[n].assessmentAnswer = "likely_down";
66  }

83  protected void SetRandomHemisphereQubit(GameObject qubit, ApplyGate script, Boolean upper)
84  {
85  // Generate random x-coordinate.
86  float x = Random.Range(-1f, 1f);
87 
88  // Generate z-coordinate based on x-coordinate such that
89  // -(sqrt(1-x^2)) <= z <= +(sqrt(1-x^2))
90  float zLimit = (float)Math.Sqrt(1 - Math.Pow(x, 2));
91  float z = Random.Range(-1 * zLimit, zLimit);
92 
93  // Generate y-coordinate based on x and z-coordinates. If it's being created in the upper hemisphere,
94  // its sign is +1, otherwise -1.
95  int sign = upper == true ? 1 : -1;
96  float y = (float)(Math.Sqrt(1 - Math.Pow(x, 2) - Math.Pow(z, 2))) * sign;
97 
98  // epsilon is making sure that y's value isn't too close to either of the up/down poles.
99  // epsilon2 is making sure that y's value isn't too close to the equator.
100  float epsilon = 0.1f, epsilon2 = 0.2f;
101 
102  // If they are, need to generate a new random value for x and re-calculate z and y coordinates
103  // until they're not too close.
104  while (1 - Math.Abs(y) < epsilon || Math.Abs(y) < epsilon2)
105  {
106  x = Random.Range(-1f, 1f);
107  zLimit = (float)Math.Sqrt(1 - Math.Pow(x, 2));
108  z = Random.Range(-1 * zLimit, zLimit);
109  y = sign * (float)(Math.Sqrt(1 - Math.Pow(x, 2) - Math.Pow(z, 2)));
111  }
112 
113  // Point the qubit with the chosen x, y, z-coordinates.
114  script.PointVector(x, z, y);
115  }
