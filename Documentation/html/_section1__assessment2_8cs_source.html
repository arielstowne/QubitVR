98  void randomizeState(int n)
99  {
101  //* Thresholds for randomizing qubits. Must sum to 100. */
102  int upChance = 5, downChance = 5, anyChance = 90;
104 
105  float choice = Random.Range(0, 100);
106 
107  // If choice is below the up thresholds, set state to up and update assessment answer to "up".
108  if (choice < upChance)
109  {
110  QubitManager.getScript(n).setState(States.UP);
111  QubitManager.getScript(n).assessmentAnswer = "up";
112  }
113 
114  // If choice is above up threshold but below down threshold, set state to down and update assessment answer to "down".
115  else if (choice < upChance + downChance)
116  {
117  QubitManager.getScript(n).setState(States.DOWN);
118  QubitManager.getScript(n).assessmentAnswer = "down";
119  }
120 
121  // If choice passes the up/down thresholds, then we use the orginal algorithm.
122  else if (choice < upChance + downChance + anyChance)
123  {
124  QubitManager.SetRandomState(n);
125  return;
126  }
